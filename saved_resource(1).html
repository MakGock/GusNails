function initCalendar() {
const calendarDays = document.getElementById('calendar-days');
const today = new Date();
const currentDate = new Date();
const currentMonth = currentDate.getMonth();
const currentYear = currentDate.getFullYear();
// Get first day of month
const firstDay = new Date(currentYear, currentMonth, 1).getDay();
// Adjust for Monday as first day (0 = Sunday, 1 = Monday, etc.)
const firstDayIndex = firstDay === 0 ? 6 : firstDay - 1;
// Get days in month
const daysInMonth = new Date(currentYear, currentMonth + 1, 0).getDate();
// Get previous month days
const prevMonthDays = new Date(currentYear, currentMonth, 0).getDate();
// Clear calendar
calendarDays.innerHTML = '';
// Previous month days (disabled)
for (let i = firstDayIndex; i > 0; i--) {
const day = document.createElement('div');
day.className = 'calendar-day text-center py-2 text-gray-400 disabled';
day.textContent = prevMonthDays - i + 1;
calendarDays.appendChild(day);
}
// Current month days
for (let i = 1; i <= daysInMonth; i++) {
const day = document.createElement('div');
day.className = 'calendar-day text-center py-2 cursor-pointer rounded-full';
day.textContent = i;
// Create date object for this day
const dayDate = new Date(currentYear, currentMonth, i);
// Highlight today
if (i === today.getDate() && currentMonth === today.getMonth() && currentYear === today.getFullYear()) {
day.classList.add('bg-pink-100', 'font-semibold');
}
// Disable past days
if (dayDate < today) {
day.classList.add('text-gray-400', 'disabled');
day.classList.remove('cursor-pointer');
}
// Add click event only for future days
if (dayDate >= today) {
day.addEventListener('click', function() {
// Remove selection from other days
document.querySelectorAll('.calendar-day').forEach(d => {
d.classList.remove('bg-pink-500', 'text-white');
});
// Select this day
this.classList.add('bg-pink-500', 'text-white');
// Update selected date display
const selectedDate = `${i}.${currentMonth + 1}.${currentYear}`;
document.getElementById('selected-date').textContent = selectedDate;
// Update booking summary
updateBookingSummary();
// Generate new time slots with buffer times
generateTimeSlots();
});
}
calendarDays.appendChild(day);
}
// Next month days (enabled)
const totalDays = firstDayIndex + daysInMonth;
const remainingDays = 7 - (totalDays % 7);
if (remainingDays < 7) {
for (let i = 1; i <= remainingDays; i++) {
const day = document.createElement('div');
day.className = 'calendar-day text-center py-2 cursor-pointer rounded-full';
day.textContent = i;
// Add click event for next month days
day.addEventListener('click', function() {
// Remove selection from other days
document.querySelectorAll('.calendar-day').forEach(d => {
d.classList.remove('bg-pink-500', 'text-white');
});
// Select this day
this.classList.add('bg-pink-500', 'text-white');
// Update selected date display (next month)
const nextMonth = currentMonth + 1;
const nextMonthYear = nextMonth > 11 ? currentYear + 1 : currentYear;
const nextMonthAdjusted = nextMonth > 11 ? 0 : nextMonth;
const selectedDate = `${i}.${nextMonthAdjusted + 1}.${nextMonthYear}`;
document.getElementById('selected-date').textContent = selectedDate;
// Update booking summary
updateBookingSummary();
// Generate new time slots with buffer times
generateTimeSlots();
});
calendarDays.appendChild(day);
}
}
// Select today by default if it's available
const todayElement = document.querySelectorAll('.calendar-day')[firstDayIndex + today.getDate() - 1];
if (todayElement && !todayElement.classList.contains('disabled')) {
todayElement.click();
} else {
// Find the first available future day
let firstAvailableDay = null;
document.querySelectorAll('.calendar-day').forEach(day => {
if (!firstAvailableDay && !day.classList.contains('disabled') && day.textContent >= today.getDate()) {
firstAvailableDay = day;
}
});
if (firstAvailableDay) {
firstAvailableDay.click();
} else {
// If no days available in current month, try next month
const nextMonthDays = document.querySelectorAll('.calendar-day');
for (let i = firstDayIndex + daysInMonth; i < nextMonthDays.length; i++) {
if (!nextMonthDays[i].classList.contains('disabled')) {
nextMonthDays[i].click();
break;
}
}
}
}
// Update month display
const monthNames = ["Январь", "Февраль", "Март", "Апрель", "Май", "Июнь", "Июль", "Август", "Сентябрь", "Октябрь", "Ноябрь", "Декабрь"];
document.getElementById('current-month').textContent = `${monthNames[currentMonth]} ${currentYear}`;
}
function generateTimeSlots() {
const timeSlotsContainer = document.getElementById('time-slots');
timeSlotsContainer.innerHTML = '';
// Generate time slots from 10:00 to 20:00 every 30 minutes
for (let hour = 10; hour <= 20; hour++) {
for (let minute = 0; minute < 60; minute += 30) {
if (hour === 20 && minute === 30) break; // Last slot at 20:00
const timeSlot = document.createElement('div');
timeSlot.className = 'time-slot text-center py-2 rounded border border-gray-200 cursor-pointer';
// Format time (add leading zero)
const formattedHour = hour.toString().padStart(2, '0');
const formattedMinute = minute.toString().padStart(2, '0');
timeSlot.textContent = `${formattedHour}:${formattedMinute}`;
// Randomly mark some slots as booked (for demo)
if (Math.random() < 0.2) {
timeSlot.classList.add('booked', 'bg-gray-100', 'text-gray-400');
timeSlot.classList.remove('cursor-pointer');
timeSlot.innerHTML += '<div class="text-xs mt-1">Занято</div>';
}
// Add buffer time after booked slots
if (Math.random() < 0.1 && !timeSlot.classList.contains('booked')) {
timeSlot.classList.add('buffer', 'bg-yellow-50', 'text-gray-400');
timeSlot.classList.remove('cursor-pointer');
timeSlot.innerHTML += '<div class="text-xs mt-1">Перерыв</div>';
}
timeSlotsContainer.appendChild(timeSlot);
}
}
}
function updateBookingSummary() {

const dateSelected = document.getElementById('summary-date').textContent !== 'Не выбрана';
const timeSelected = document.getElementById('summary-time').textContent !== 'Не выбрано';
// Update summary date
document.getElementById('summary-date').textContent = document.getElementById('selected-date').textContent;
// Enable/disable confirm button
const confirmButton = document.getElementById('confirm-booking');
if (masterSelected && serviceSelected && dateSelected && timeSelected) {
confirmButton.disabled = false;
} else {
confirmButton.disabled = true;
}
}
